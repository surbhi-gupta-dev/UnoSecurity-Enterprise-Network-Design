!Bonus Configurations

!1. Guest network should have access to web server only.
!Configuring on Router(BO)

enable
config t
ip access-list extended GUEST-WEB-ONLY
 permit tcp 192.168.0.0 0.0.1.255 host 172.16.3.100 eq 80
 permit tcp 192.168.0.0 0.0.1.255 host 172.16.3.100 eq 443
 deny ip 192.168.0.0 0.0.1.255 any
 exit
 
interface FastEthernet0/1.30
 description Guest-Network to Head Office
 ip access-group GUEST-WEB-ONLY in
 exit

!2.	Only 192.168.3.10 host from the HR network can access the account server.
!Configuring on Router (BO)

enable
config t
access-list 101 permit ip host 192.168.3.10 host 172.16.1.100

interface FastEthernet0/1.20
ip access-group 101 in
exit

!3. Only Account network can access the Account Server
!Configuring MultiLayered Switch 0

enable
config t
access-list 102 permit ip host 192.168.3.10 host 172.16.1.100
access-list 102 permit ip 172.16.1.0 0.0.0.255 host 172.16.1.100

interface vlan 40
ip access-group 102 out
exit

!4. Only IT network can access the DNS Server
!Configuring MultiLayered Switch 0

enable
config t
access-list 103 permit tcp 172.16.2.0 0.0.0.255 host 172.16.3.100 eq 53
access-list 103 permit udp 172.16.2.0 0.0.0.255 host 172.16.3.100 eq 53
access-list 103 permit tcp 192.168.0.0 0.0.1.255 host 172.16.3.100 eq 80
access-list 103 permit tcp 192.168.0.0 0.0.1.255 host 172.16.3.100 eq 443
access-list 103 deny any host 172.16.3.100
access-list 103 permit any any 

interface vlan 60
ip access-group 103 in
exit
